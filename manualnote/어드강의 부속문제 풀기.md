#컨테이너 운반
def bubble(k):
    for i in range(1,len(k)):
        if k[i]>k[i-1]:
            k[i],k[i-1]=k[i-1],k[i]
            while k[i-1]>k[i-2]:
                if i<2:
                    break
                else:
                    k[i-1],k[i-2]=k[i-2],k[i-1]
                    i-=1

for tc in range(int(input())):
    n,m=map(int,input().split())
    nweight=list(map(int,input().split()))
    mload=list(map(int,input().split()))
    bubble(nweight)
    bubble(mload)
    ans=0
    for i in range(m):
        for j in range(n):
            if mload[i]>=nweight[j]:
                ans+=nweight[j]
                n-=1
                nweight.pop(j)
                break
    print('#{} {}'.format(tc+1,ans))

#병합정렬(원래대로 구현하면 조건 미충족)
def devide(m):
    global tt
    if len(m)==1:
        return m
    elif len(m)==2:
        if m[0] > m[1]:
            tt += 1
            m.sort()
        return m
    k=len(m)//2
    m1=m[:k]
    m2=m[k:]
    m1=devide(m1)
    m2=devide(m2)
    if m1[-1]>m2[-1]:
        tt+=1
    new=sorted(sum([m1,m2],[]))
    return new

for tc in range(int(input())):
    n=int(input())
    raw=list(map(int,input().split()))
    tt=0
    ans=devide(raw)
    print('#{} {} {}'.format(tc+1,ans[len(ans)//2],tt))
