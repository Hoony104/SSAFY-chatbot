def boom(i,j,brics):
    global ans
    cnt=brics[i][j]
    if cnt==0:
        return
    brics[i][j]=0
    if cnt==1:
        ans+=1
        return
    else:
        ans+=1
        for q in range(cnt):
            if i-q>0:
                boom(i-q,j,brics)
            if i+q<h:
                boom(i+q,j,brics)
            if j-q>0:
                boom(i,j-q,brics)
            if j+q<w:
                boom(i,j+q,brics)

def drop(n,brics):
    for i in range(h):
        if brics[i][n]!=0:
            boom(i,n,brics)
            break
