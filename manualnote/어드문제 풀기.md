```python
            k-=temp
            bi[i]=1
```

```python
#최소합.  DP로 짜기
for tc in range(int(input())):
    n=int(input())
    matrix=[list(map(int,input().split())) for _ in range(n)]
    accmat=[[0]*n for _ in range(n)]
    accmat[0][0]=matrix[0][0]
    for i in range(1,n):
        accmat[i][0]=accmat[i-1][0]+matrix[i][0]
        accmat[0][i]=accmat[0][i-1]+matrix[0][i]
    for i in range(1,n):
        for j in range(1,n):
            accmat[i][j]=min(accmat[i-1][j],accmat[i][j-1])+matrix[i][j]
    real=accmat[n-1][n-1]
    print('#{} {}'.format(tc+1,real))
```

```python
#컨테이너 운반
def bubble(k):
    for i in range(1,len(k)):
        if k[i]>k[i-1]:
            k[i],k[i-1]=k[i-1],k[i]
            while k[i-1]>k[i-2]:
                if i<2:
                    break
                else:
                    k[i-1],k[i-2]=k[i-2],k[i-1]
                    i-=1

for tc in range(int(input())):
    n,m=map(int,input().split())
    nweight=list(map(int,input().split()))
    mload=list(map(int,input().split()))
    bubble(nweight)
    bubble(mload)
    ans=0
    for i in range(m):
        for j in range(n):
            if mload[i]>=nweight[j]:
                ans+=nweight[j]
                n-=1
                nweight.pop(j)
                break
    print('#{} {}'.format(tc+1,ans))
```

```python
def babygin(m):
    card=[0]*10
    for i in m:
        card[i]+=1
    for i in range(10):
        if card[i]>=3:
            return True
    for i in range(8):
        if card[i]>=1 and card[i+1]>=1 and card[i+2]>=1:
            return True
```
